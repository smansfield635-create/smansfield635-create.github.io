import worldKernel from "./world/world_kernel.js"; import planetEngine from "./world/planet_engine.js"; import runtime from "./world/runtime.js"; const BOOT_META = Object.freeze({ name: "index", version: "G1", role: "boot_orchestrator", deterministic: true, sourceOfTruth: false }); const BOOT_CONSTANTS = Object.freeze({ REQUIRED_LATTICE_SIZE: 16, REQUIRED_TOTAL_NODES: 256, REQUIRED_SEQUENCE: Object.freeze(["kernel", "engine", "runtime"]) }); const deepFreeze = (value) => { if (value === null || typeof value !== "object" || Object.isFrozen(value)) return value; Object.getOwnPropertyNames(value).forEach((key) => { deepFreeze(value[key]); }); return Object.freeze(value); }; const hasOwn = (obj, key) => Object.prototype.hasOwnProperty.call(obj, key); const assert = (condition, code) => { if (!condition) { const error = new Error(code); error.code = code; throw error; } }; const validateKernel = (kernel) => { assert(kernel && typeof kernel === "object", "KERNEL_MISSING"); assert(Object.isFrozen(kernel), "KERNEL_NOT_IMMUTABLE"); assert(hasOwn(kernel, "meta"), "KERNEL_META_MISSING"); assert(hasOwn(kernel, "constants"), "KERNEL_CONSTANTS_MISSING"); assert(hasOwn(kernel, "lattice"), "KERNEL_LATTICE_MISSING"); assert(hasOwn(kernel, "validation"), "KERNEL_VALIDATION_MISSING"); assert(kernel.meta && kernel.meta.sourceOfTruth === true, "KERNEL_NOT_SOURCE_OF_TRUTH"); assert(kernel.meta && kernel.meta.immutable === true, "KERNEL_IMMUTABLE_FLAG_INVALID"); assert(kernel.meta && kernel.meta.deterministic === true, "KERNEL_DETERMINISM_FLAG_INVALID"); assert(kernel.meta && kernel.meta.latticeSize === BOOT_CONSTANTS.REQUIRED_LATTICE_SIZE, "KERNEL_LATTICE_SIZE_INVALID"); assert(kernel.meta && kernel.meta.totalNodes === BOOT_CONSTANTS.REQUIRED_TOTAL_NODES, "KERNEL_TOTAL_NODES_INVALID"); assert(Array.isArray(kernel.lattice.nodes), "KERNEL_LATTICE_NODES_INVALID"); assert(kernel.lattice.nodes.length === BOOT_CONSTANTS.REQUIRED_TOTAL_NODES, "KERNEL_LATTICE_COVERAGE_INVALID"); assert(kernel.validation && kernel.validation.ok === true, "KERNEL_VALIDATION_FAILED"); return Object.freeze({ ok: true, immutable: Object.isFrozen(kernel), latticeCoverage: kernel.lattice.nodes.length, deterministic: kernel.meta.deterministic === true }); }; const validateEngine = (engine, kernel) => { assert(engine && typeof engine === "object", "ENGINE_MISSING"); assert(Object.isFrozen(engine), "ENGINE_NOT_IMMUTABLE"); assert(engine !== kernel, "ENGINE_EQUALS_KERNEL"); assert(!((engine.meta && engine.meta.sourceOfTruth === true) || engine.sourceOfTruth === true), "ENGINE_DEFINES_TRUTH"); const coverage = typeof engine.totalNodes === "number" ? engine.totalNodes : engine.meta && typeof engine.meta.totalNodes === "number" ? engine.meta.totalNodes : kernel.lattice.nodes.length; assert(coverage === BOOT_CONSTANTS.REQUIRED_TOTAL_NODES, "ENGINE_LATTICE_COVERAGE_INVALID"); return Object.freeze({ ok: true, immutable: Object.isFrozen(engine), latticeCoverage: coverage, deterministic: engine.meta ? engine.meta.deterministic !== false : true }); }; const validateRuntime = (runtimeModule) => { assert(runtimeModule && typeof runtimeModule === "object", "RUNTIME_MISSING"); assert(Object.isFrozen(runtimeModule), "RUNTIME_NOT_IMMUTABLE"); assert(!((runtimeModule.meta && runtimeModule.meta.sourceOfTruth === true) || runtimeModule.sourceOfTruth === true), "RUNTIME_DEFINES_TRUTH"); assert(!((runtimeModule.meta && runtimeModule.meta.mutatesState === true) || runtimeModule.mutatesState === true), "RUNTIME_MUTATES_STATE"); return Object.freeze({ ok: true, immutable: Object.isFrozen(runtimeModule), deterministic: runtimeModule.meta ? runtimeModule.meta.deterministic !== false : true }); }; const validateModules = (kernel, engine, runtimeModule) => { const kernelValidation = validateKernel(kernel); const engineValidation = validateEngine(engine, kernel); const runtimeValidation = validateRuntime(runtimeModule); return Object.freeze({ ok: kernelValidation.ok && engineValidation.ok && runtimeValidation.ok, kernel: kernelValidation, engine: engineValidation, runtime: runtimeValidation }); }; const bindPipeline = (kernel, engine, runtimeModule) => Object.freeze({ sequence: BOOT_CONSTANTS.REQUIRED_SEQUENCE, kernel, engine, runtime: runtimeModule }); const createSystemHandle = (pipeline, validation) => deepFreeze({ meta: BOOT_META, validation, pipeline: Object.freeze({ sequence: pipeline.sequence }), runtime: pipeline.runtime, getKernelMeta: () => pipeline.kernel.meta, getEngineMeta: () => (pipeline.engine && pipeline.engine.meta ? pipeline.engine.meta : null), getRuntimeMeta: () => (pipeline.runtime && pipeline.runtime.meta ? pipeline.runtime.meta : null), getValidation: () => validation, getSequence: () => pipeline.sequence.slice() }); const initialize = () => { const kernel = worldKernel; const engine = planetEngine; const runtimeModule = runtime; const validation = validateModules(kernel, engine, runtimeModule); assert(validation.ok === true, "BOOT_VALIDATION_FAILED"); const pipeline = bindPipeline(kernel, engine, runtimeModule); return createSystemHandle(pipeline, validation); }; const systemHandle = initialize(); export { systemHandle, initialize }; export default systemHandle;
