} function buildReceiptChain() { return deepFreeze({ kernel: worldKernel.getReceipt(), engine: planetEngine.getReceipt(), runtime: runtime.getReceipt(), render: render.getReceipt() }); } function buildProjectionSummary() { const out = {};
